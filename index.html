/** * Chatbots Entre Cerros - tickets -> mail/ para extension Trigger Email. * Base Firestore: chatbotentrecerros */ const { onDocumentCreated } = require('firebase-functions/v2/firestore'); const { defineString } = require('firebase-functions/params'); const { initializeApp } = require('firebase-admin/app'); const { getFirestore, FieldValue } = require('firebase-admin/firestore'); const DB = 'chatbotentrecerros'; const DEFAULT_CONDOMINIO = 'entre-cerros'; const DEFAULT_ADMIN_EMAIL = 'conserjeria.wm815@gmail.com'; const DEFAULT_PUBLIC_APP_URL = 'https://contactosadm-svg.github.io/Chatbots-Entre-Cerros/'; const PRIMARY_COLOR = '#0d9488'; const deployAdminEmail = defineString('ADMIN_EMAIL', { default: DEFAULT_ADMIN_EMAIL }); initializeApp(); function db() { return getFirestore(DB); } function escapeHtml(value) { return String(value ?? '') .replace(/&/g, '&') .replace(//g, '>') .replace(/"/g, '"'); } function parseEmails(raw) { return String(raw || '') .split(/[,;]/) .map((email) => email.trim()) .filter(Boolean); } function formatTicketDate(ticket) { const ts = ticket.fechaCreacion || ticket.createdAt; const date = typeof ts === 'number' ? new Date(ts) : ts?.toDate ? ts.toDate() : new Date(ts || Date.now()); return date.toLocaleString('es-CL', { timeZone: 'America/Santiago' }); } function normalizeAttachment(att) { if (!att) return null; return { name: att.name || att.nombre || 'adjunto', url: att.url || '' }; } function row(label, value) { return ` ${escapeHtml(label)} ${escapeHtml(value || '-')} `; } function buildTicketEmail({ ticketId, condominioId, ticket, settings }) { const buildingName = settings.buildingName || 'Entre Cerros'; const adminPanelUrl = settings.adminPanelUrl || DEFAULT_PUBLIC_APP_URL; const primary = settings.primaryColor || PRIMARY_COLOR; const fecha = formatTicketDate(ticket); const adjuntos = (ticket.adjuntos || []).map(normalizeAttachment).filter(Boolean); const adjuntosRows = adjuntos.length ? adjuntos.map((att) => { const label = escapeHtml(att.name); const link = att.url ? `${label}` : label; return `${link}`; }).join('') : 'Sin archivos adjuntos'; const subject = `[${buildingName}] Nuevo ticket ${ticketId} - ${ticket.categoria || 'Solicitud'}`; const html = `

${escapeHtml(buildingName)}

Nueva solicitud de atencion

Ticket
${escapeHtml(ticketId)}
${row('Nombre', ticket.nombre)} ${row('Departamento', ticket.departamento)} ${row('Correo', ticket.email)} ${row('Categoria', ticket.categoria || ticket.asunto)} ${row('Prioridad', ticket.prioridad)} ${row('Estado', ticket.estado || 'Abierto')} ${row('Fecha', fecha)}

Descripcion

${escapeHtml(ticket.descripcion || '-')}

Adjuntos

${adjuntosRows}
Abrir panel de tickets

${escapeHtml(buildingName)} - ${escapeHtml(condominioId)} - ARTURITO AI

`; const text = [ `Nueva solicitud: ${ticketId}`, `Edificio: ${buildingName}`, `Fecha: ${fecha}`, `Nombre: ${ticket.nombre || '-'}`, `Departamento: ${ticket.departamento || '-'}`, `Correo: ${ticket.email || '-'}`, `Categoria: ${ticket.categoria || '-'}`, `Prioridad: ${ticket.prioridad || '-'}`, '', 'Descripcion:', ticket.descripcion || '-', '', adjuntos.length ? `Adjuntos: ${adjuntos.map((a) => a.url || a.name).join(', ')}` : 'Sin adjuntos', '', `Panel: ${adminPanelUrl}` ].join('\n'); return { subject, html, text }; } function buildTicketStatusUrl(ticketId) { return `${DEFAULT_PUBLIC_APP_URL}/?ticket=${encodeURIComponent(ticketId)}`; } function buildClientConfirmationEmail({ ticketId, ticket, settings }) { const buildingName = settings.buildingName || 'Entre Cerros'; const primary = settings.primaryColor || PRIMARY_COLOR; const fecha = formatTicketDate(ticket); const statusUrl = buildTicketStatusUrl(ticketId); const subject = `[${buildingName}] Confirmacion solicitud ${ticketId}`; const html = `

${escapeHtml(buildingName)}

Tu solicitud fue registrada

Hola ${escapeHtml(ticket.nombre || 'residente')}, recibimos tu solicitud y el equipo de administracion la revisara pronto.

Numero de ticket
${escapeHtml(ticketId)}

Puedes consultar el estado en cualquier momento con tu numero de ticket.

Consultar estado

${escapeHtml(buildingName)} - ARTURITO AI

`; const text = [ `Confirmacion de solicitud - ${buildingName}`, '', `Hola ${ticket.nombre || 'residente'},`, '', 'Tu solicitud fue registrada correctamente.', `Numero de ticket: ${ticketId}`, `Fecha: ${fecha}`, '', `Consulta el estado: ${statusUrl}` ].join('\n'); return { subject, html, text, statusUrl }; } async function resolveRecipients(settings, condominioId) { const requiredAdmin = parseEmails(DEFAULT_ADMIN_EMAIL)[0]; let to = requiredAdmin; let cc = parseEmails(settings.notifyCcEmail); if (to) cc = cc.filter((email) => email.toLowerCase() !== to.toLowerCase()); if (to) return { to: [to], cc }; try { const snap = await db().doc(`condominios/${condominioId}/email_config/outbound`).get(); if (snap.exists) { const data = snap.data() || {}; to = parseEmails(data.notifyAdminEmail || data.adminEmail || data.to)[0] || requiredAdmin; cc = parseEmails(data.notifyCcEmail || data.cc); if (to) cc = cc.filter((email) => email.toLowerCase() !== to.toLowerCase()); if (to) return { to: [to], cc }; } } catch (err) { console.warn('[email] email_config read:', err.message); } to = parseEmails(deployAdminEmail.value() || process.env.ADMIN_EMAIL || DEFAULT_ADMIN_EMAIL)[0] || requiredAdmin; return { to: to ? [to] : [], cc: [] }; } async function ensureMailQueued({ recipients, cc = [], email, ticketId, condominioId, source, ticket }) { const mailRef = db().collection('mail').doc(ticketId); const snap = await mailRef.get(); if (snap.exists) return ticketId; await mailRef.set({ to: recipients[0], ...(cc.length ? { cc: cc.length === 1 ? cc[0] : cc } : {}), replyTo: email.replyTo, message: { subject: email.subject, html: email.html, text: email.text }, createdAt: FieldValue.serverTimestamp(), status: 'pending', meta: { tipo: 'ticket_created', ticketId, condominioId, source: source || 'function', database: DB, createdAt: FieldValue.serverTimestamp(), ticketSnapshot: ticket ? { nombre: ticket.nombre || '', departamento: ticket.departamento || '', email: ticket.email || '', categoria: ticket.categoria || '' } : null } }); return ticketId; } async function ensureClientConfirmationQueued({ ticket, ticketId, condominioId, settings, source }) { const clientEmail = parseEmails(ticket.email)[0]; if (!clientEmail) return null; const mailId = `${ticketId}-client`; const mailRef = db().collection('mail').doc(mailId); const snap = await mailRef.get(); if (snap.exists) return mailId; const email = buildClientConfirmationEmail({ ticketId, ticket, settings }); await mailRef.set({ to: clientEmail, replyTo: parseEmails(settings.smtpReplyTo || settings.notifyAdminEmail || DEFAULT_ADMIN_EMAIL)[0] || undefined, message: { subject: email.subject, html: email.html, text: email.text }, createdAt: FieldValue.serverTimestamp(), status: 'pending', meta: { tipo: 'ticket_confirmation', ticketId, condominioId, source: source || 'function', database: DB, statusUrl: email.statusUrl, createdAt: FieldValue.serverTimestamp() } }); return mailId; } exports.notifyAdminOnTicket = onDocumentCreated( { document: 'condominios/{condominioId}/tickets/{ticketId}', database: DB, region: 'southamerica-west1' }, async (event) => { const ticket = event.data?.data(); if (!ticket) return; const { condominioId, ticketId } = event.params; if (condominioId !== DEFAULT_CONDOMINIO) return; const settingsSnap = await db().doc(`condominios/${condominioId}/settings/main`).get(); const settings = settingsSnap.exists ? settingsSnap.data() : {}; const recipients = await resolveRecipients(settings, condominioId); if (!recipients.to.length) { console.error(`[notifyAdminOnTicket] Sin destinatarios (${condominioId})`); return; } const email = buildTicketEmail({ ticketId, condominioId, ticket, settings }); email.replyTo = parseEmails(ticket.email)[0] || settings.smtpReplyTo || DEFAULT_ADMIN_EMAIL; const mailId = await ensureMailQueued({ recipients: recipients.to, cc: recipients.cc, email, ticketId, condominioId, source: 'notifyAdminOnTicket', ticket }); console.log(`[notifyAdminOnTicket] mail/${mailId} encolado`); const clientMailId = await ensureClientConfirmationQueued({ ticket, ticketId, condominioId, settings, source: 'notifyAdminOnTicket' }); if (clientMailId) console.log(`[notifyAdminOnTicket] mail/${clientMailId} confirmacion encolada`); } );
